
\begin{resumeBox}
  \emph{À retenir dans une semaine :} 
  \begin{niceitemize}
    \item Un espace vectoriel est un ensemble avec deux lois : une addition interne et une multiplication par un scalaire.
    \item Un vecteur est un élément d'un espace vectoriel.
    \item Pour vérifier qu'un ensemble est un espace vectoriel  :
      \begin{itemize}
        \item[$\bullet$] Possède-t-il un vecteur nul ?
        \item[$\bullet$] L'addition est-elle une loi interne ?
        \item[$\bullet$] La multiplication par un scalaire est-elle une loi externe ?
        \item[$\bullet$] Les règles de calcul de l'addition et de la multiplication par un scalaire sont-elles vérifiées ?
      \end{itemize}
  \end{niceitemize}
\end{resumeBox}

\begin{rappelsBox}
  \begin{niceitemize}
    \item Que signifie qu'une loi soit interne à un ensemble ?
    \item Quelles sont les règles de calculs pour les espaces vectoriels ?
    \item Qu'est-ce que le vecteur nul d'un espace vectoriel ?
  \end{niceitemize}
\end{rappelsBox}

  \ifthenelse{\boolean{showSolutions}}{

\subsection{parties de $\mathbb{R}^2$}

  Les parties suivantes sont-elles des sous-espaces vectoriels de $\mathbb{R}^2$ ?

  \ifthenelse{\boolean{showSolutions}}{}{\begin{multicols}{2} }
  \begin{enumerate}
    \item $A=\left\{(x, y) \in \mathbb{R}^2 \mid x \leqslant y\right\}$ 

    \ifthenelse{\boolean{showSolutions}}{
      \vspace{1em}
      \begin{mdframed}
        $A$ est un demi-plan de $\mathbb{R}^2$, ce n'est pas un espace vectoriel, si on prend un vecteur non nul et qu'on multiplie par un scalaire négatif, on obtient un vecteur qui n'est pas dans $A$.
      \end{mdframed}
    }{}
    \vspace{1em}

    \item $B=\left\{(x, y) \in \mathbb{R}^2 \mid x y=0\right\}$ 

    \ifthenelse{\boolean{showSolutions}}{
      \vspace{1em}
      \begin{mdframed}
        $B$ est une réunion de deux droites qui passent par l'origine. Si on prend un vecteur sur chaque droite et qu'on les additionne, on obtient un vecteur qui n'est pas dans $B$.
      \end{mdframed}
    }{}
    \vspace{1em}

    \item $C=\left\{(x, y) \in \mathbb{R}^2 \mid x=y\right\}$ 

    \ifthenelse{\boolean{showSolutions}}{
      \vspace{1em}
      \begin{mdframed}
        $C$ est une droite qui passe par l'origine. C'est un espace vectoriel.
      \end{mdframed}
    }{}
    \vspace{1em}

    \item $D=\left\{(x, y) \in \mathbb{R}^2 \mid x+y=1\right\}$

    \ifthenelse{\boolean{showSolutions}}{
      \vspace{1em}
      \begin{mdframed}
        Le vecteur nul n'est pas dans $D$, donc $D$ n'est pas un espace vectoriel.
      \end{mdframed}
    }{}
    \vspace{1em}

  \end{enumerate}
  \ifthenelse{\boolean{showSolutions}}{}{\end{multicols} }

  \subsection{Dans un espace de fonctions}

  Soit $E$ le $\mathbb{R}$-espace vectoriel des applications de $[0,1]$ dans $\mathbb{R}$ muni des opérations usuelles. \newline 
  Soit $F$ l'ensemble des applications de $[0,1]$ dans $\mathbb{R}$ vérifiant l'une des conditions suivantes. \newline 
  Dans quel cas $F$ est-il un espace vectoriel inclus dans $E$?


  \ifthenelse{\boolean{showSolutions}}{}{\begin{multicols}{2} }
  \begin{enumerate}
    \item $f(0)+f(1)=0$

    \ifthenelse{\boolean{showSolutions}}{
      \vspace{1em}
      \begin{mdframed}
        C'est bien un espace vectoriel, pour le montrer on répond aux 3 questions : 
        \begin{itemize}
          \item Le vecteur nul est dans $F$, c'est la fonction nulle.
          \item L'addition est une loi interne, pour tout $f,g \in F$, $f+g \in F$.
          \item La multiplication par un scalaire est une loi externe, pour tout $f \in F$ et $\lambda \in \mathbb{R}$, $\lambda f \in F$.
        \end{itemize}
      \end{mdframed}
    }{}
    \vspace{1em}

    \item $f(0)=0$

    \ifthenelse{\boolean{showSolutions}}{
      \vspace{1em}
      \begin{mdframed}
        C'est bien un espace vectoriel, pour le montrer on répond aux 3 questions : 
        \begin{itemize}
          \item Le vecteur nul est dans $F$, c'est la fonction nulle.
          \item L'addition est une loi interne, pour tout $f,g \in F$, $f+g \in F$.
          \item La multiplication par un scalaire est une loi externe, pour tout $f \in F$ et $\lambda \in \mathbb{R}$, $\lambda f \in F$.
        \end{itemize}
      \end{mdframed}
    }{}
    \vspace{1em}

    \item $f\left(\frac{1}{2}\right)=\frac{1}{4}$

    \ifthenelse{\boolean{showSolutions}}{
      \vspace{1em}
      \begin{mdframed}
        Le vecteur nul n'est pas dans $F$, donc $F$ n'est pas un espace vectoriel.
      \end{mdframed}
    }{}
    \vspace{1em}

    \item $\forall x \in[0,1], f(x)+f(1-x)=0$

    \ifthenelse{\boolean{showSolutions}}{
      \vspace{1em}
      \begin{mdframed}
        C'est bien un espace vectoriel, on le montre en détaillant les 3 questions :
        \begin{itemize}
          \item Le vecteur nul est dans $F$, c'est la fonction nulle.
          \item L'addition est une loi interne, pour tout $f,g \in F$, $f+g \in F$.
          \item La multiplication par un scalaire est une loi externe, pour tout $f \in F$ et $\lambda \in \mathbb{R}$, $\lambda f \in F$.
        \end{itemize}
      \end{mdframed}
    }{}
    \vspace{1em}

    \item $\forall x \in[0,1], f(x) \geqslant 0$

    \ifthenelse{\boolean{showSolutions}}{
      \vspace{1em}
      \begin{mdframed}
        $F$ n'est pas un espace vectoriel, si on prend la fonction $f(x)=x$ et qu'on la multiplie par un scalaire négatif, on obtient une fonction qui n'est pas dans $F$.
      \end{mdframed}
    }{}
    \vspace{1em}

    \item $2 f(0)=f(1)+3$

    \ifthenelse{\boolean{showSolutions}}{
      \vspace{1em}
      \begin{mdframed}
        $F$ n'est pas un espace vectoriel, la fonction nulle n'est pas dans $F$.
      \end{mdframed}
    }{}
  \end{enumerate}
  \ifthenelse{\boolean{showSolutions}}{}{\end{multicols} }

\subsection{Dans $\mathbb{R}[X]$}


Les ensembles suivants sont-ils des espaces vectoriels?
\begin{enumerate}
  \item $A_1$ est l'ensemble des polynômes réels P vérifiant $\mathrm{P}(0)=1$.

  \ifthenelse{\boolean{showSolutions}}{
    \vspace{1em}
    \begin{mdframed}
      $A_1$ ne contient pas le polynome nul, donc $A_1$ n'est pas un espace vectoriel.
    \end{mdframed}
  }{}
  \vspace{1em}

  \item $A_2$ est l'ensemble des polynômes réels ayant $a$ comme racine. ( $a \in \mathbb{R}$ fixé).

  \ifthenelse{\boolean{showSolutions}}{
    \vspace{1em}
    \begin{mdframed}
      $A_2$ est bien un espace vectoriel.
    \end{mdframed}
  }{}
  \vspace{1em}

  \item $A_3$ est l'ensemble des polynômes réels ayant au moins une racine réelle.

  \ifthenelse{\boolean{showSolutions}}{
    \vspace{1em}
    \begin{mdframed}
      $A_3$ n'est pas un espace vectoriel, on peut trouver deux polynômes de $A_3$ dont la somme n'est pas dans $A_3$.\newline 
      Par exemple, $X^2$ et $(X-1)^2$ sont dans $A_3$, mais $X^2+(X-1)^2$ n'est pas dans $A_3$ car ce polynôme n'admet aucune racine. 
    \end{mdframed}
  }{}
  \vspace{1em}

  \item $A_4$ est l'ensemble des polynômes réels de degré 3.

  \ifthenelse{\boolean{showSolutions}}{
    \vspace{1em}
    \begin{mdframed}
      Exemple du cours : $A_4$ n'est pas un espace vectoriel, l'addition n'est pas une loi interne.
    \end{mdframed}
  }{}
  \vspace{1em}

\end{enumerate}
