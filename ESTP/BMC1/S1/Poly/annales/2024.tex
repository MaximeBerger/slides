\section{Exponentielles et logarithmes}


\begin{enumerate}
    \item 
    Calculer $\left(e^4\times e^{-1}\right)^5$.

    \item
    Exprimer $\ln(16)-\ln(2^{-8})+\ln(2) $ en fonction de $\ln(2)$.

    \item 
    Résoudre $\ln(2x+1)>4$.
        \item 
    Résoudre $e^{5x-2}=1$.
      \item 
    Résoudre $e^{5x}=-1$.
\end{enumerate}




\section{Nombres complexes}

Considérons les nombres complexes $z_1= 1+i$, $z_2=-4i$, $z_3=6i$ et $z_4=1-i\sqrt{3}$.

\begin{enumerate}
    \item 
    Calculer les conjugués de $z_1$, $z_2$, $z_3$ et $z_4$.
    \item
    Calculer $z_1\times z_4$.
    \item
    Donner la forme algébrique de $\dfrac{z_1}{z_4}$.
\end{enumerate}
